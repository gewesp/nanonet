# 1-dimensional, 2 elements
{ 1 2 }

# 2-dimensional, 1x2
{ { 1 2 } }

# 2-dimensional, 2x1
{ 
  { 1 }
  { 2 }
}

# Same as above
{ { 1 } { 2 } }

# 3-dimensional, 3x2x1
{ 
  { { 1 } { 2 } }
  { { 3 } { 4 } }
  { { 5 } { 6 } }
}

# 3-dimensional, 1x2x3
{
  {
    { 1 2 3 }
    { 4 5 6 }
  }
}

# 3-dimensional, 3x1x3
{
  {
    { 1 2 3 }
  }
  {
    { 4 5 6 }
  }
  {
    { 7 8 9 }
  }
}
